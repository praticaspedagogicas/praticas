async function readLiveMoment(sessionCode=state.settings.sessionCode,{silent=true}={}){

function enqueueLiveMoment(target,{silent=false,reason='manual'}={}){
 const sessionCode=state.settings.sessionCode;
 if(!sessionCode){if(!silent)toast('Selecione primeiro a sessão.');return Promise.resolve(false)}

 desiredLiveMoment=target;
 participationBusy=true;
 renderParticipationControls();
 const operation=async()=>{
   try{
     const row=await backend().setMoment({
       sessionCode,
       controlKey:controlKey(),
       runId:state.settings.runId||'legacy',
       moment:target,
       sessionTitle:state.settings.sessionTitle||'',
       sessionDate:state.settings.date||null
     });
     if(state.settings.sessionCode===sessionCode){
       liveMoment=row?.is_open?(row.active_moment||target):'aguardar';
       desiredLiveMoment=liveMoment;
       liveStateKnown=true;
       lastParticipationError='';
     }
     if(!silent&&reason==='manual')toast(target==='aguardar'?'Participação fechada.':'Participação aberta.');
     return true;
   }catch(e){
     console.error('Falha ao alterar participação no Supabase',e);
     lastParticipationError=e?.message||'erro desconhecido';
     try{await readLiveMoment(sessionCode,{silent:true})}catch(_e){}
     if(!silent)toast(`Não foi possível ${target==='aguardar'?'fechar':'abrir'} a participação: ${lastParticipationError}`);
     return false;
   }finally{
     if(state.settings.sessionCode===sessionCode){participationBusy=false;renderParticipationControls()}
   }
 };
 const queued=liveMutationQueue.then(operation,operation);

}
